/**
 * 将纯 base64 字符串转换为 File 对象
  * @param base64 - base64 编码的字符串（不含 data URL 前缀）
   * @param mimeType - 文件 MIME 类型，如 'image/jpeg'
    * @param fileName - 文件名
     */
     export function base64ToFile(base64: string, mimeType: string, fileName: string = 'file'): File {
        const binaryString = atob(base64);
          const bytes = new Uint8Array(binaryString.length);
            for (let i = 0; i < binaryString.length; i++) {
                  bytes[i] = binaryString.charCodeAt(i);
            }
              return new File([bytes], fileName, { type: mimeType });
     }

     /**
      * 将 data URL 格式的 base64 转换为 File 对象
       * @param dataUrl - 完整的 data URL 字符串
        * @param fileName - 文件名（可选）
         * @returns File 对象
          */
          export function dataURLToFile(dataUrl: string, fileName: string = 'image.png'): File {
              const arr = dataUrl.split(',');
                const mime = arr[0].match(/:(.*?);/)?.[1] || 'image/png';
                  const bstr = atob(arr[1]);
                    let n = bstr.length;
                      const u8arr = new Uint8Array(n);
                        while (n--) {
                              u8arr[n] = bstr.charCodeAt(n);
                        }
                          return new File([u8arr], fileName, { type: mime });
          }
                        }
          }
            }
     }
